def first_k_primes(n):
	guess = 2
	output_list = [guess]
	while len(output_list) < n:
		is_prime = True
		guess += 1
		for j in range(2, (guess // 2) + 1, 1):
			if guess % j == 0:
				is_prime = False		
		if is_prime:
			output_list.append(guess)
	return(output_list)
